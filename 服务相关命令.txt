#3.启动服务
python window_monitor.py start
window_monitor.exe start

#4.重启服务
python window_monitor.py restart

#5.停止服务
python window_monitor.py stop

#6.删除/卸载服务
python window_monitor.py remove
window_monitor.exe remove
